240 {

242  closeProject();
243 
244  auto file = std::make_shared<QFile>(filename);
245  QFileInfo fileinfo {*file};
246 
247  file->open(QIODevice::ReadOnly);
248 
249  //-- try to read project from xml file
250  ProjectStorageXML storage_xml (file);
251 
252  //-- subscribe to the reader's events

256

261 
262  p_project->setDirty(false);
263 
264  cryEventSys(
265  MsgLevel::INFO,
266  tr("Project \"") + filename + tr("\" opened successfully")
267  );
268 
269  rememberProjectFilename(fileinfo.absoluteFilePath());
270 
271  connectToCurProject();
272 
273  } catch (std::invalid_argument e){

276  cryEventSys(

288 void Appl::saveProject(QString filename)
289 {
290  //-- if given filename is empty, use current project filename
291  if (filename.isEmpty()){
292  filename = getProjectFilename();
293  }
294 
295  //-- if filename is, after all, specified, then proceed

299  QFileInfo fileinfo {*file};
300 
301  file->open(QIODevice::WriteOnly);
302 
303  //-- try to read project from xml file

305

313  cryEventSys(

325  cryEventSys(
