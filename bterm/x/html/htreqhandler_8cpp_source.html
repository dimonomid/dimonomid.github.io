116 ReqHandler::Result ReqHandler::handle(
117  QJSValue input_msg_jsval,
118  QJSValue script_ctx_jsval
119  )
120 {
121  QJSValue factory_jsval = p_engine->newQObject(p_script_factory.get());
122  QQmlEngine::setObjectOwnership(p_script_factory.get(), QQmlEngine::CppOwnership);
123 
124  //-- before handling, set global properties
125  p_engine->globalObject().setProperty("LITTLE_END", ByteArrRead::LITTLE_END);
126  p_engine->globalObject().setProperty("BIG_END", ByteArrRead::BIG_END);
127  p_engine->globalObject().setProperty("factory", factory_jsval);
128 
129  ReqHandler::Result ret = Result::UNKNOWN;
130 
131  //-- before running the handler, set response to an empty pointer.
132  p_response = std::shared_ptr<ByteArrReadWrite>();
133  last_error = Error::UNKNOWN;
134  last_exception_details = QVariantMap();
135 
136  QJSValue result;
137 
138 
139 
140  //-- try to evaluate JavaScript code
141  QJSValue func = p_engine->evaluate(script_func_code);
142 
143  if (func.isError()){
144  //-- evaluation is failed: some error in JavaScript code.
145 
146  ret = Result::ERROR;
147  last_error = Error::EXCEPTION;
148  last_exception_details = MyUtil::qjsErrorToVariant(func);
149  qDebug() << "exception 1: " << last_exception_details;
150 
151  } else if (!func.isCallable()){
152  //-- evaluation has succeed, but the script has evaluated to
153  // non-function value.
154 
155  ret = Result::ERROR;
156  last_error = Error::SCRIPT_IS_NOT_FUNCTION;
157 
158  } else {
159  //-- ok, we have function value. Try to call it,
160  // passing the scripts context as `this`.
161  QJSValue returned = func.callWithInstance(
162  script_ctx_jsval,
163  QJSValueList() << input_msg_jsval
164  );
165 
166  if (returned.isError()){
167  //-- returned value is an error: some exception has happened
168  // during handler invocation
169 
170  ret = Result::ERROR;
171  last_error = Error::EXCEPTION;
172  last_exception_details = MyUtil::qjsErrorToVariant(returned);
173  qDebug() << "exception 2: " << last_exception_details;
174  } else {
175  //-- ok, handler has executed successfully. Let's see the result:
176  // whether the request was handled or not.
177 
178  bool handled = returned.property("handled").toBool();
179  if (handled){
180  //-- yes, request is handled
181  ret = Result::OK_HANDLED;
182 
183  if (returned.property("response").isQObject()){
184  //-- we have "response" property, which, if defined, should
185  // contain response to send. So, remember it: the client
186  // may want to get it by calling `getResponse()`.
187 
188  p_response = std::shared_ptr<ByteArrReadWrite>(
189  dynamic_cast<ByteArrReadWrite *>(
190  returned.property("response").toQObject()
191  )
192  );
193  //-- don't forget to set ownership of this object to
194  // C++ code, so that JavaScript engine won't
195  // garbage-collect it.
196  QQmlEngine::setObjectOwnership(
197  p_response.get(),
198  QQmlEngine::CppOwnership
199  );
200 
201  //-- p_response now contains the byte array that was
202  // written by the script
203  } else {
204  //-- although the request is handled, we have no response.
205  // Ok, response is already set to an empty pointer, so,
206  // leave it as it is.
207  }
208 
209  } else {
210  //-- request was not handled. That's ok.
211  ret = Result::OK_NOT_HANDLED;
212  }
213 
214  }
215 
216  }
217 
218  return ret;
219 }
