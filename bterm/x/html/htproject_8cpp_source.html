169 /**
170  * Called when new Rx message is decoded
171  */
172 void Project::onMessageDecoded(const DataMsg &msg)
173 {
174  //qDebug(msg.toString().c_str());
175 
176  auto p_event = std::make_shared<EventDataMsg>(msg, EventDataMsg::Direction::RX);
177  emit (eventDataMsg(p_event));
178 
179  std::shared_ptr<std::vector<uint8_t>> p_req_data = msg.getUserData();
180 
181  //-- create an object that will be given to handlers as input message
182  QJSValue input_msg_jsval = p_engine->newObject();
183 
184  //-- actual input byte array
185  ByteArrRead ba_in {*p_req_data};
186  QJSValue ba_in_jsval = p_engine->newQObject(&ba_in);
187  QQmlEngine::setObjectOwnership(&ba_in, QQmlEngine::CppOwnership);
188 
189  input_msg_jsval.setProperty("byteArr", ba_in_jsval);
190 
191  //-- iterate through all the request handlers
192  for (auto p_req_handler : handlers){
193 
194  //-- try to handle the request with current handler
195  ReqHandler::Result res = p_req_handler->handle(
196  input_msg_jsval,
197  script_ctx_jsval
198  );
199 
200  //-- take action depending on handling result
201  switch (res){
202  case ReqHandler::Result::UNKNOWN:
203  qDebug("unknown result: should never be here");
204  break;
205 
206  case ReqHandler::Result::OK_NOT_HANDLED:
207  //-- this handler did not handle the request.
208  // Ok, we will proceed to the next handler (if any)
209  break;
210 
211  case ReqHandler::Result::OK_HANDLED:
212  //-- request is handled
213  {
214  //-- get response, encode it and transmit on the wire
215  auto p_data_tx = p_req_handler->getResponse();
216  if (p_data_tx->size() > 0){
217  DataMsg msg_tx = p_codec->encodeMessage(*p_data_tx);
218  auto p_data_raw_tx = msg_tx.getRawData();
219  p_io_dev->write(*p_data_raw_tx);
220 
221  //-- emit an event about outgoing (Tx) message
222  auto p_event = std::make_shared<EventDataMsg>(
223  msg_tx,
224  EventDataMsg::Direction::TX,
225  p_req_handler
226  );
227  emit (eventDataMsg(p_event));
228  } else {
229  //-- no response is generated. That's ok, so, we don't
230  // send anything, just stop iterating through
231  // handlers.
232  }
233  }
234  break;
235 
236  case ReqHandler::Result::ERROR:
237  //-- some error has happened during executing the handler.
238  // (this is error in JavaScript handler code)
239  qDebug() << "error";
240  break;
241  }
242 
243  if (res == ReqHandler::Result::OK_HANDLED){
244  //-- if request was handled, then stop iterating handlers,
245  // and exit immediately
246  break;
247  }
248  }
249 }
