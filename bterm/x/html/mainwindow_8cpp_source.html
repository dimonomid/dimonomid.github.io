200 bool MainWindow::event(QEvent *e)
201 {
202  bool ret = QMainWindow::event(e);
203 
204  if (e->type() == QEvent::WindowStateChange){
205  //-- window was just maximized or minimized.
206  // If it is minimized, then we should restore geometry that window
207  // had last time when it was minimized.
208  //
209  // All this mess is needed because size of central widget (the log)
210  // is not saved as part of the state, and when we restore
211  // maximized window, we should set the geometry to occupy the whole
212  // screen.
213 
214  auto p_sett = appl.settings();
215  if (!isMaximized() && need_to_restore_non_maximized_geometry){
216  QRect rect = p_sett->value(SETT_KEY__MAINWINDOW__GEOMETRY).toRect();
217  if (rect.width() != 0 && rect.height() != 0){
218  setGeometry( rect );
219  }
220  need_to_restore_non_maximized_geometry = false;
221  }
222 
223  }
224 
225  return ret;
226 }
