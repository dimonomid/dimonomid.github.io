150  QJSValue input_msg_jsval

152 {
153  //-- before handling, set global properties
154 
155  ReqHandler::Result ret = Result::UNKNOWN;
156 
157  last_error = Error::UNKNOWN;
158  last_exception_details = QVariantMap();
159 
160  QJSValue result;
161

163 
164  //-- check if we already have evaluated handler function
165  if (!handler_func.isCallable()){
166  //-- no, so, try to evaluate JavaScript code
167  handler_func = p_jshost->evaluate(
168  script_func_code,
169  this->title
170  );
171  }
172 
173  if (handler_func.isError()){
174  //-- evaluation is failed: some error in JavaScript code.
175 
176  ret = Result::ERROR;
177  last_error = Error::EXCEPTION;
178  last_exception_details = MyUtil::qjsErrorToVariant(handler_func);
179  qDebug() << "exception 1: " << last_exception_details;
180 
181  } else if (!handler_func.isCallable()){
182  //-- evaluation has succeed, but the script has evaluated to
183  // non-function value.
184 
185  ret = Result::ERROR;
186  last_error = Error::SCRIPT_IS_NOT_FUNCTION;
187 
188  } else {
189 
190  //-- ok, we have function value. Try to call it,
191  // passing the scripts context as `this`.
192  QJSValue returned = p_jshost->callFuncWithCommonContext(
193  handler_func,
194  QJSValueList() << input_msg_jsval,
195  this->title
196  );
197 
198 
199  if (returned.isError()){
200  //-- returned value is an error: some exception has happened
201  // during handler invocation
202 
203  ret = Result::ERROR;
204  last_error = Error::EXCEPTION;
205  last_exception_details = MyUtil::qjsErrorToVariant(returned);
206  qDebug() << "exception 2: " << last_exception_details;
207  } else {
208  //-- ok, handler has executed successfully. Let's see the result:
209  // whether the request was handled or not.
210 
211  bool handled = returned.property("handled").toBool();
212  ret = handled ? Result::OK_HANDLED : Result::OK_NOT_HANDLED;
213 
214  }
215 
216  }
