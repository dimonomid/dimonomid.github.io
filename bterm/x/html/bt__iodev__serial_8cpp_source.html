154 void IODevSerial::onReadyRead()
155 {
156  //qDebug("ready read");
157 
158  uint32_t avail_size = 0;
159 
160  while ((avail_size = p_port->bytesAvailable()) > 0){
161  size_t existing_size = cur_data.size();
162  cur_data.resize( existing_size + avail_size );
163  uint8_t *p_data = cur_data.data();
164  char *p_data_char = reinterpret_cast<char *>(p_data);
165 
166  uint32_t read_size = p_port->read(
167  p_data_char + existing_size, avail_size
168  );
169 
170  //-- if we actually read less than was declared as "available",
171  // adjust cur_data size
172  if (read_size < avail_size){
173  cur_data.resize( existing_size + read_size );
174  } else if (read_size > avail_size){
175  //-- should never be here
176  qDebug("error: read more than was available");
177  }
178  }
179 
180  emit readyRead(cur_data.size());
181 }
