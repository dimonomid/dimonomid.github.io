117 ReqHandler::Result Project::runHandler(
118  std::shared_ptr<ReqHandler> p_handler,
119  QJSValue input_msg_jsval
120  )
121 {
122  //-- try to handle the request with current handler
123  ReqHandler::Result res = p_handler->handle(
124  input_msg_jsval
125  );
126 
127  //-- take action depending on handling result
128  switch (res){
129  case ReqHandler::Result::UNKNOWN:
130  qDebug("unknown result: should never be here");
131  break;
132 
133  case ReqHandler::Result::OK_NOT_HANDLED:
134  //-- this handler did not handle the request.
135  // Ok, we will proceed to the next handler (if any)
136  break;
137 
138  case ReqHandler::Result::OK_HANDLED:
139  //-- request is handled
140  break;
141 
142  case ReqHandler::Result::ERROR:
143  //-- some error has happened during executing the handler.
144  // (this is error in JavaScript handler code)
145  qDebug() << "error";
146  break;
147  }
148 
149  return res;
150 }

216 void Project::setIODev(std::shared_ptr<IODev> p_io_dev)
217 {
218  //-- if we already have some IODev set, then unsubscribe from its events
219  if (this->p_io_dev != nullptr){
220  if (this->p_io_dev->isOpened()){
221  this->p_io_dev->close();
222  }
223 
224  disconnect(
225  this->p_io_dev.get(), &IODev::readyRead,
226  this, &Project::onIODevReadyRead
227  );
228  disconnect(
229  this->p_io_dev.get(), &IODev::baudRateChanged,
230  this, &Project::onIODevBaudRateChanged
231  );
232 
233  }
234 
235  this->p_io_dev = p_io_dev;
236 
237  //-- get existing received data and discard it
238  this->p_io_dev->read();
239 
240  connect(
241  this->p_io_dev.get(), &IODev::readyRead,
242  this, &Project::onIODevReadyRead
243  );
244 
245  //-- when baudrate of IODev changes, we want to keep our copy up-to-date
246  connect(
247  this->p_io_dev.get(), &IODev::baudRateChanged,
248  this, &Project::onIODevBaudRateChanged
249  );
250 
251  //-- check if we baudrate of IODev differs from the needed one
252  if (p_io_dev->getBaudRate() != this->baudrate){
253  //-- it differs; we need to change it.
254 
255  if (!this->p_io_dev->isOpened()){
256  //-- ok, port is not opened, so, set baudrate and open it.
257  this->p_io_dev->setBaudRate(this->baudrate);
258  this->p_io_dev->open();
259  } else {
260  //-- port is already opened, so we can't alter its baudrate.
261  // For now, just echo warning. In the future, we may to close it,
262  // set baudrate, and open it again. The difficulty is that this
263  // should be done asynchronously (when device is actually closed,
264  // the signal `openStatusChanged()` is emitted)
265  auto p_event = std::make_shared<EventSys>(
266  MsgLevel::WARNING,
267  QString("IO device is opened when setting it to project. ")
268  + QString("Please change baudrate manually to ")
269  + QString::number(this->baudrate)
270  );
271  emit event(p_event);
272  }
273  } else if (!this->p_io_dev->isOpened()){
274  this->p_io_dev->open();
275  }
276 }
